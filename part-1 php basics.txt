<?php
    function returnSeconLargestNumber($numArr) {
        if (count($numArr) < 2) {
            return null; 
        }


        $first = $second = PHP_INT_MIN;


        foreach ($numArr as $num) {
            if ($num > $first) {
                $second = $first;
                $first = $num;
            } elseif ($num > $second && $num < $first) {
                $second = $num;
            }
        }


        return ($second === PHP_INT_MIN) ? null : $second; 
    }
?>
